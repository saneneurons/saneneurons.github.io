JavaScript Code Sample

						

function getRandomInt(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min; // excluding the max
}

getRandomInt(10, 20)

Python Code Sample

						

import random

def getRandom(min, max):
	n = random.randint(min,max) # including the max
	return n

print(getRandom(10, 20))
